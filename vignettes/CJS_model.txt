model {
  phi[1] <- 1
  p[1] <- 1
  for (j in 2:J) {
    phi[j] ~ dbeta(1, 1)
    p[j] ~ dbeta(1, 1)
  }
  for (i in 1:N) {
    for (j in (f[i] + 1):J) {
      z[i, j] ~ dbern(phi[j] * z[i, j - 1])
      y[i, j] ~ dbern(p[j] * z[i, j])
    }
  }
  survship[1] <- 1
  for (j in 2:J) {
    survship[j] <- survship[j - 1] * phi[j]
  }
}
